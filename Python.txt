def JsonFile():
    new_path = 'C:/Users/toc/Desktop/Python_Codes/Data/Input'
    name = askopenfilename(initialdir="C:/Users/toc/Desktop/Python_Codes/Data/JSON_File",
                           filetypes=(("Json File", "*.json"), ("All Files", "*.*")),
                           title="Choose a file."
                           )
    try:
        with open(name,'r', encoding='utf8') as infile:
            o = json.load(infile)
            chunkSize = 1
        os.chdir(new_path)
        json_files = os.path.basename(name)
        for i in range(0, len(o), chunkSize):
            with open(json_files + '_' + str(i//chunkSize) + '.json', 'w') as outfile:
                json.dump(o[i:i+chunkSize], outfile)
                
    except:
        messagebox.showwarning("Warning", "Select a file")
